--This is the main query to execute.
SELECT table_name, dsql2('select count(*) from '||table_name) as rownum
FROM information_schema.tables
WHERE table_type='BASE TABLE'
    AND table_schema='public'
ORDER BY 2 DESC;

--This is the supporting function. Need to create this function first.
This just takes a string command, executes it and returns the value. Probably only works for single row resulting queries.
CREATE OR REPLACE FUNCTION dsql2(i_text text)
  RETURNS int AS
$BODY$
Declare
  v_val int;
BEGIN
  execute i_text into v_val;
  return v_val;
END;
$BODY$
  LANGUAGE plpgsql VOLATILE
  COST 100;
